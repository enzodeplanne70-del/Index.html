    <script>
        let app = { n: "ANTHONY", on: false, m: 0 };
        
        const specs = [
            { w: 42, h: 130, s: 45, b: 25, p: 0.8, mega: false },
            { w: 58, h: 180, s: 95, b: 45, p: 0.9, mega: false },
            { w: 80, h: 230, s: 170, b: 80, p: 1.0, mega: true }
        ];

        function saveName() {
            const val = document.getElementById('name-input').value;
            if(!val) return;
            app.n = val;
            document.getElementById('display-name').textContent = app.n.toUpperCase();
            document.getElementById('scr-init').classList.add('hide');
            document.getElementById('scr-menu').classList.remove('hide');
        }

        function startTorch(idx) {
            app.m = idx;
            app.on = false;
            const sp = specs[idx];
            const obj = document.getElementById('torch-model');
            obj.style.width = sp.w + 'px';
            obj.style.height = sp.h + 'px';
            document.getElementById('scr-menu').classList.add('hide');
            document.getElementById('scr-play').classList.remove('hide');
            render();
        }

        function quit() { 
            app.on = false; 
            render(); 
            document.getElementById('scr-play').classList.add('hide'); 
            document.getElementById('scr-menu').classList.remove('hide'); 
        }

        document.getElementById('torch-btn').onclick = () => {
            app.on = !app.on;
            if(window.navigator.vibrate) window.navigator.vibrate(app.on ? 70 : 30);
            render();
        };

        function render() {
            const system = document.getElementById('light-system');
            const beam = document.getElementById('light-beam');
            const glow = document.getElementById('light-glow');
            const flash = document.getElementById('mega-flash');
            const btn = document.getElementById('torch-btn');
            const sp = specs[app.m];

            if(app.on) {
                btn.classList.add('active');
                system.style.opacity = "1";
                const color = 'rgba(255,255,255,'+sp.p+')';
                
                beam.style.background = `conic-gradient(from -${sp.s/2}deg at bottom center, transparent ${50-(sp.s/6)}%, ${color} 50%, transparent ${50+(sp.s/6)}%)`;
                beam.style.filter = `blur(${sp.b}px)`;
                glow.style.background = `radial-gradient(circle, ${color} 0%, transparent 80%)`;
                
                if(sp.mega) flash.style.opacity = "0.75";
            } else {
                btn.classList.remove('active');
                system.style.opacity = "0";
                flash.style.opacity = "0";
            }
        }

        const updatePos = (x) => {
            if(!app.on) return;
            const rot = (x - window.innerWidth/2) / 12;
            document.getElementById('light-system').style.transform = `rotate(${rot}deg)`;
        };
        document.addEventListener('mousemove', (e) => updatePos(e.clientX));
        document.addEventListener('touchmove', (e) => updatePos(e.touches[0].clientX));
    </script>
